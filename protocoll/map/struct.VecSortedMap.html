pub struct VecSortedMap<K, V>(_);

an array-map sorted by key.

does not support entry; see Map::update for the same functionality. MapMut functions are not much more efficient than Map for this data structure.

this makes up for the (intended) absence of iter_mut.

example

// a somewhat unecessary way to create a mapping from square numbers to
// fibonacci numbers.
use protocoll::map::VecSortedMap;
use protocoll::MapMut;
let mut m:VecSortedMap<u32,u32> =
    (0..).map(|n| (n * n, 0))
    .take(13).collect();
let (ref mut a, ref mut b) = (0,1);
m.update_all_mut
    (|_,v| {*v = *a;
            *a = *b;
            *b += *v});
assert_eq!(m[&   0], 0);
assert_eq!(m[&   1], 1);
assert_eq!(m[&   4], 1);
assert_eq!(m[&   9], 2);
assert_eq!(m[&  16], 3);
assert_eq!(m[&  25], 5);
assert_eq!(m[&  36], 8);
assert_eq!(m[& 144], 144);
